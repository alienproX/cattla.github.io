import SwiftUI

struct BorderView: View {
  
  private var paddingLeft: CGFloat = 0
  private var height: CGFloat = 0
  private var color: Color = Color._tableBorder
  
  init(_ paddingLeft: CGFloat = 0, color: Color = Color._tableBorder, height: CGFloat = 0.5) {
      self.paddingLeft = paddingLeft
      self.height = height
      self.color = color
  }
  
  var body: some View {
      Rectangle()
          .fill(color)
          .frame(maxWidth: .infinity, minHeight: self.height, maxHeight: self.height, alignment: .center)
          .padding(.leading, self.paddingLeft)
  }
}
